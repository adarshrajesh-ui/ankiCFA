const homeMap = document.getElementById("home-map");
const homeMapField = document.getElementById("home-map-field");
const homeMapNote = document.getElementById("home-map-note");
const homeMapNodes = [...document.querySelectorAll("#home-map-field .node, #home-map-field .subnode")];
let mapState = { x: 0, y: 0, scale: 1 };
let pinchStart = null;

function clamp(value, min, max) {
  return Math.max(min, Math.min(max, value));
}

function updateMapTransform() {
  homeMapField.style.transform = `translate(${mapState.x}px, ${mapState.y}px) scale(${mapState.scale})`;
}

function updateNearestConcept() {
  const box = homeMap.getBoundingClientRect();
  const center = { x: box.left + box.width / 2, y: box.top + box.height / 2 };
  let nearest = homeMapNodes[0];
  let nearestDistance = Infinity;

  for (const node of homeMapNodes) {
    const nodeBox = node.getBoundingClientRect();
    const nodeCenter = { x: nodeBox.left + nodeBox.width / 2, y: nodeBox.top + nodeBox.height / 2 };
    const distance = Math.hypot(nodeCenter.x - center.x, nodeCenter.y - center.y);
    if (distance < nearestDistance) {
      nearest = node;
      nearestDistance = distance;
    }
  }

  homeMapNodes.forEach((node) => node.classList.toggle("active", node === nearest));
  const name = nearest.dataset.name || "Concept";
  const detail = nearest.dataset.detail || "Priority and cards due appear here.";
  homeMapNote.innerHTML = `<b>Nearest: ${name}</b>${detail}`;
}

if (homeMap && homeMapField && homeMapNote) {
  updateMapTransform();
  updateNearestConcept();

  homeMap.addEventListener("wheel", (event) => {
    event.preventDefault();
    mapState.x = clamp(mapState.x - event.deltaX * 0.38, -96, 96);
    mapState.y = clamp(mapState.y - event.deltaY * 0.30, -78, 78);
    mapState.scale = clamp(mapState.scale + Math.abs(event.deltaY) * 0.0009, 1, 1.38);

    updateMapTransform();
    updateNearestConcept();
  }, { passive: false });

  homeMap.addEventListener("mouseleave", () => {
    mapState.scale = Math.max(1, mapState.scale - 0.06);
    updateMapTransform();
    updateNearestConcept();
  });

  homeMap.addEventListener("touchstart", (event) => {
    if (event.touches.length === 2) {
      const [a, b] = event.touches;
      pinchStart = {
        distance: Math.hypot(a.clientX - b.clientX, a.clientY - b.clientY),
        scale: mapState.scale,
        x: mapState.x,
        y: mapState.y,
        centerX: (a.clientX + b.clientX) / 2,
        centerY: (a.clientY + b.clientY) / 2
      };
    }
  }, { passive: true });

  homeMap.addEventListener("touchmove", (event) => {
    if (event.touches.length === 2 && pinchStart) {
      event.preventDefault();
      const [a, b] = event.touches;
      const distance = Math.hypot(a.clientX - b.clientX, a.clientY - b.clientY);
      const centerX = (a.clientX + b.clientX) / 2;
      const centerY = (a.clientY + b.clientY) / 2;
      const ratio = distance / Math.max(1, pinchStart.distance);

      mapState.scale = clamp(pinchStart.scale * ratio, 0.74, 1.55);
      mapState.x = clamp(pinchStart.x + (centerX - pinchStart.centerX) * 0.42, -112, 112);
      mapState.y = clamp(pinchStart.y + (centerY - pinchStart.centerY) * 0.42, -88, 88);

      updateMapTransform();
      updateNearestConcept();
    }
  }, { passive: false });

  homeMap.addEventListener("touchend", (event) => {
    if (event.touches.length < 2) {
      pinchStart = null;
    }
  });

  homeMapNodes.forEach((node) => {
    node.addEventListener("mouseenter", () => {
      homeMapNodes.forEach((candidate) => candidate.classList.toggle("active", candidate === node));
      const name = node.dataset.name || "Concept";
      const detail = node.dataset.detail || "Need and due status appear here.";
      homeMapNote.innerHTML = `<b>${name}</b>${detail}`;
    });

    node.addEventListener("click", () => {
      const name = node.dataset.name || "Concept";
      const detail = node.dataset.detail || "Cards due for this concept.";
      homeMapNodes.forEach((candidate) => candidate.classList.toggle("active", candidate === node));
      homeMapNote.innerHTML = `<b>Cards Due: ${name}</b>${detail} · Opening filtered due-card queue for this section.`;
    });
  });
}

function queueDirection(event){
  event.preventDefault();
  const form = event.currentTarget;
  const direction = new FormData(form).get("direction");
  if(!direction) return;
  if(window.lavish && window.lavish.queuePrompt){
    window.lavish.queuePrompt("Home Screen direction: " + direction, {
      tag: "home-screen",
      text: "Home Screen: " + direction,
      element: form,
      queueKey: "home-screen-direction",
      data: { direction }
    });
  }
}
